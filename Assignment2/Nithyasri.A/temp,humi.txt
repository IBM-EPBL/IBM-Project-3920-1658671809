import requests
a="https://api.openweathermap.org/data/2.5/weather?q=Chennai,IN&appid=1628081571520a0fe87e778a934c53c4"
r=requests.get(url=a)
data=r.json()
print(data)
temp=data['main']['temp']
humidity =data['main']['humidity']
print("temperature:" , temp)
print("humidity:" , humidity)
if temp>100:
    print("Alarm ON")
else:
    print("Alarm OFF")
